# Update package repositories
sudo apt update

# 1. Core Python and Build Tools
sudo apt install -y python3-pip python3-dev python3-venv git build-essential

# 2. Kivy GUI & Graphics Libraries (SDL2 + OpenGL)
sudo apt install -y \
    libgl1-mesa-dev libgles2-mesa-dev libegl1-mesa-dev \
    libsdl2-dev libsdl2-image-dev libsdl2-mixer-dev libsdl2-ttf-dev \
    libmtdev-dev xclip xsel

# 3. GStreamer Media Framework (for Screensaver Video Playback)
sudo apt install -y \
    libgstreamer1.0-dev libgstreamer-plugins-base1.0-dev \
    gstreamer1.0-plugins-base gstreamer1.0-plugins-good \
    gstreamer1.0-plugins-bad gstreamer1.0-plugins-ugly gstreamer1.0-libav

# 4. Image Processing & OpenCV Libraries
sudo apt install -y libopencv-dev python3-opencv libjpeg-dev zlib1g-dev

# 5. Smartcard Reader Service (ACR122U RFID reader support)
sudo apt install -y pcscd pcsc-tools libpcsclite-dev swig
sudo systemctl enable pcscd
sudo systemctl start pcscd




sudo bash -c "echo 'blacklist pn533' >> /etc/modprobe.d/blacklist.conf"
sudo bash -c "echo 'blacklist nfc' >> /etc/modprobe.d/blacklist.conf"
